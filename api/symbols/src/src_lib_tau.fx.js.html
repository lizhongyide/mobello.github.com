1522     var solveCubicBezierFunction = function(p1x, p1y, p2x, p2y, t, duration) {
1523       function sampleCurveX(t) {
1524         // `ax t^3 + bx t^2 + cx t' expanded using Horner's rule.
1525         return ((ax * t + bx) * t + cx) * t;
1526       }
1527       function sampleCurveY(t) {
1528         return ((ay * t + by) * t + cy) * t;
1529       }
1530       function sampleCurveDerivativeX(t) {
1531         return (3.0 * ax * t + 2.0 * bx) * t + cx;
1532       }
1533       // Given an x value, find a parametric value it came from.
1534       function solveCurveX(x, epsilon) {
1535         var t0, t1, t2, x2, d2, i;
1536         function fabs(n) {
1537           if (n >= 0) {
1538             return n;
1539           } else {
1540             return 0 - n;
1541           }
1542         }
1543         // First try a few iterations of Newton's method -- normally very fast.
1544         for (t2 = x, i = 0; i < 8; i++) {
1545           x2 = sampleCurveX(t2) - x;
1546           if (fabs(x2) < epsilon) {
1547             return t2;
1548           }
1549           d2 = sampleCurveDerivativeX(t2);
1550           if (fabs(d2) < 1e-6) {
1551             break;
1552           }
1553           t2 = t2 - x2 / d2;
1554         }
1555         // Fall back to the bisection method for reliability.
1556         t0 = 0.0;
1557         t1 = 1.0;
1558         t2 = x;
1559         if (t2 < t0) {
1560           return t0;
1561         }
1562         if (t2 > t1) {
1563           return t1;
1564         }
1565         while (t0 < t1) {
1566           x2 = sampleCurveX(t2);
1567           if (fabs(x2 - x) < epsilon) {
1568             return t2;
1569           }
1570           if (x > x2) {
1571             t0 = t2;
1572           } else {
1573             t1 = t2;
1574           }
1575           t2 = (t1 - t0) * .5 + t0;
1576         }
1577         // Failure.
1578         return t2;
1579       }
1580       function solve(x, epsilon) {
1581         return sampleCurveY(solveCurveX(x, epsilon));
1582       }

1584       // The epsilon value we pass to UnitBezier::solve given that the animation
1585       // is going to run over |dur| seconds. The longer the animation, the more
1586       // precision we need in the timing function result to avoid ugly
1587       // discontinuities.
1588       function solveEpsilon(duration) {
1589         return 1.0 / (200.0 * duration);
1590       }
1591 
1592       // Calculate the polynomial coefficients, implicit first and last control
1593       // points are (0,0) and (1,1).
1594       var ax = 0, bx = 0, cx = 0, ay = 0, by = 0, cy = 0;
1595       cx = 3.0 * p1x;
1596       bx = 3.0 * (p2x - p1x) - cx;
1597       ax = 1.0 - cx - bx;
1598       cy = 3.0 * p1y;
1599       by = 3.0 * (p2y - p1y) - cy;
1600       ay = 1.0 - cy - by;
1601 
1602       return solve(t, solveEpsilon(duration));
1603     };
