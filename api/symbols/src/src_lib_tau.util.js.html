  31       /**
  32        * default 값을 목적 객체(dest)에 적용한다.
  33        * @example
  34        * tau.util.applyDefault({foo: 'bar'}, {aaa: 'bbb', ccc: 'ddd'});
  35        * // returns {"foo":"bar","aaa":"bbb","ccc":"ddd"}
  36        *
  37        * @param {Object} dest 목적 객체
  38        * @param {Object} defaultObj default객체
  39        * @returns {Object} Default값이 적용된 객체
  40        */
  41       applyDefault: function(dest, defaultObj, deep) {
  42         var c, p;
  43         if (dest == null && !deep) {
  44           dest = defaultObj;
  45           return dest;
  46         }
  47         if ((typeof defaultObj) == "object") {
  48           c = defaultObj.constructor;
  49           switch (c) {
  50           case Object:
  51             dest = dest||{};
  52             break;
  53           case Array:
  54             dest = dest||[];
  55             break;
  56           default:
  57             dest = defaultObj;
  58             return dest;
  59           }
  60           for (p in defaultObj) {
  61             if (dest[p] === undefined) {
  62               if (deep) {
  63                 dest[p] = tau.util.applyDefault(undefined, defaultObj[p], deep);
  64               } else {
  65                 dest[p] = defaultObj[p];
  66               }
  67             }
  68           }
  69         } else {
  70           dest = defaultObj;
  71         }
  72         return dest;
  73       },

  75       /**
  76        * 명시된 HTML 앨리먼트에 스타일을 설정하거나, value가
  77        * 명시되지 않으면 해당 스타일 이름에 해당하는 값을
  78        * 반환한다.
  79        * @param {HTMLElement} elem HTML 앨리먼트
  80        * @param {String} name 스타일 이름
  81        * @param {String} value 스타일 값
  82        * @returns {String} value가 명시되지 않았을 경우
  83        * style이름에 해당하는 값
  84        */
  85       style: function(elem, name, value) {
  86         if (!tau.isElement(elem) || !name) return undefined;
  87 
  88         if ((name === "width" || name === "height") && parseFloat(value) < 0) {
  89           value = undefined;
  90         }
  91 
  92         var style = elem.style, 
  93             set = value !== undefined;
  94 
  95         name = tau.camelize(name);
  96         
  97         if (set) style[name] = value;
  98 
  99         return style[name];
 100       },

 102       /**
 103        * 명시된 앨리먼트의 computed style 값을 반환한다.
 104        * @param {HTMLElement} elem HTML 앨리먼트
 105        * @param {String} name 스타일 이름
 106        * @param {Boolean} [force = false] false일 경우 현재
 107        * style에 명시된 name있을 경우 그 값을 반환한다.
 108        * @returns {String|Number} computed style 값
 109        */
 110       getComputedStyle: function(elem, name, force) {
 111         var style = elem.style, val, currentStyle;
 112 
 113         if (!force && style && style[name]) {
 114           val = style[name];
 115         } else if (name === "width" || name === "height") {
 116           val = (name === "width") ? elem.clientWidth : elem.clientHeight;
 117           if (val == 0) {
 118             var prevStyle = {},
 119               props = {
 120                 position: "absolute",
 121                 visibility: "hidden",
 122                 display: "block"
 123               };
 124 
 125             for ( var key in props) {
 126               prevStyle[key] = style[key];
 127               style[key] = props[key];
 128             }
 129             val = (name === "width") ? elem.clientWidth : elem.clientHeight;
 130 
 131             for ( var key in props) {
 132               style[key] = prevStyle[key];
 133             }
 134           }
 135           val = Math.max(0, Math.round(val));
 136 
 137         } else if (document.defaultView && document.defaultView.getComputedStyle) {
 138           var defaultView = elem.ownerDocument.defaultView;
 139   
 140           if (!defaultView) return null; 
 141   
 142           var computedStyle = defaultView.getComputedStyle(elem, null);
 143 
 144           name = tau.underscore(name).replace(/_/g, '-');
 145 
 146           if (computedStyle) val = computedStyle.getPropertyValue(name); 
 147 
 148           if (name === "opacity" && val === "") val = "1"; 
 149 
 150         } else if (currentStyle = elem.currentStyle) {
 151           val = currentStyle[name] || currentStyle[tau.camelize(name)];
 152   
 153           if (/^-?(?:\d*\.)?\d+(?!px)[^\d\s]+$/i.test(val)) {
 154             style = elem.style;
 155             var left = style.left,
 156               runtimeStyle = elem.runtimeStyle,
 157               rsLeft = runtimeStyle.left;
 158             
 159             runtimeStyle.left = currentStyle.left;
 160             style.left = (name === "font-size") ? "1em" : (val || 0);
 161             val = style.pixelLeft + "px";
 162 
 163             style.left = left;
 164             runtimeStyle.left = rsLeft;
 165           }
 166         }
 167         return val;
 168       },

 170       /**
 171        * 명시된 쿠기값을 설정한다.
 172        * @param {String} name 쿠키 이름
 173        * @param {String} value 쿠키 값
 174        * @param {Number} [days] 쿠키가 유지될 기간(날자로 입력). 만약 명시되지
 175        * 않으면 세션이 종료됨과 동시에 쿠키값이 사라진다.
 176        */
 177       setCookie: function (name, value, days) {
 178         var date, expires = '';
 179         if (days) {
 180           date = new Date();
 181           date.setTime(date.getTime() + (days * 24 * 3600 * 1000));
 182           expires = "; expires=" + date.toGMTString();
 183         }
 184         document.cookie = name.concat('=', encodeURI(value), expires, '; path=/');
 185       },

 187       /**
 188        * 명시된 이름에 해당하는 쿠키값을 반환한다.
 189        * @param {String} name 쿠키 이름
 190        */
 191       getCookie: function (name) {
 192         var cookie,
 193             delimiter = name.concat('='),
 194             cookies = document.cookie.split(';');
 195         for (var i = cookies.length - 1; i >= 0; i--) {
 196           cookie = cookies[i].replace(/^\s+/, ''); // LTrim
 197           if (cookie.indexOf(delimiter) == 0) {
 198             return decodeURI(cookie.substring(delimiter.length));
 199           }
 200         }
 201         return null;
 202       },

 221     /**
 222      * 명시된 HTML앨리먼트에 CSS className이 설정되었는지 확인하고
 223      * 그 결과를 반환한다.
 224      * @param {HTMLElement} element HTML DOM 앨리먼트
 225      * @param {String} className CSS클래스 이름
 226      * @param {String} [prefix] prefix 문자열
 227      * @returns {Boolean} 명시된 className이 설정되어 있다면
 228      * true를 반환한다.
 229      */
 230     hasClass: function(element, className, prefix) {
 231       if (element) {
 232         var classList = element.classList;
 233         className = (prefix || '') + className;
 234         if (classList){
 235           return classList.contains(className);
 236         } else {
 237           var elementClassName = element.className;
 238           return (elementClassName.length > 0 && (elementClassName == className || new RegExp(
 239               "(^|\\s)" + className + "(\\s|$)").test(elementClassName)));
 240         }
 241       }
 242       return false;
 243     },

 245     /**
 246      * 명시된 HTML앨리먼트에 className을 설정한다.
 247      * @param {HTMLElement} element HTML DOM 앨리먼트
 248      * @param {String} className CSS 클래스 이름
 249      * @param {String} [prefix] prefix 문자열
 250      * @returns {Boolean} 성공적으로 CSS class가 설정되었다면
 251      * true를 반환한다.
 252      */
 253     addClass: function(element, className, prefix) {
 254       // Tag the DOM node as an EventDelegator
 255       if (!element || !className) {
 256         return false;
 257       }
 258       className = (prefix || '') + className;
 259       
 260       var classList = element.classList;
 261       if (classList){
 262         classList.add(className);
 263         return true;
 264       } else {
 265         if (!tau.util.dom.hasClass(element, className)) {
 266           element.className += (element.className ? ' ' : '') + className;
 267           return true;
 268         }
 269       }
 270       return false;
 271     },

 308     replaceClass: function(element, oldC, newC, prefix) {
 309       if (element){
 310         if (oldC) {
 311           oldC = (prefix || '') + oldC;
 312         }
 313         if (newC){
 314           newC = (prefix || '') + newC;
 315         }
 316         
 317         var classList = element.classList;
 318         if (classList){
 319           if (oldC){
 320             classList.remove(oldC);
 321           }
 322           classList.add(newC);
 323         } else {
 324           if (tau.util.dom.hasClass(element, oldC)) {
 325             element.className = element.className.replace(
 326                 new RegExp("(^|\\s+)" + oldC + "(\\s+|$)"), ' ' +newC).replace(
 327                     /^\s+/, '').replace(/\s+$/, '');
 328           } else {
 329             tau.util.dom.addClass(element, newC);
 330           }
 331         }
 332         return true;
 333       }
 334       return false;

 382     /** 
 383      * 명시된 HTML 앨리먼트를 부모 노드로 부터 삭제한다.
 384      * 삭제할 앨리먼트를 comma(,)로 분리하여 복수개를 지정할
 385      * 수 있다.
 386      * @param {HTMLElement|Array} elem 삭제할 DOM앨리먼트, 1개 이상의 
 387      * 파라미터를 기술할 수 있다.
 388      * @returns {Object|Array} 삭제된 HTML DOM앨리먼트
 389      */
 390     removeElements: function (elem) {
 391       var i, e, r, elems, removed; 
 392       if (arguments.length === 1 && elem.parentNode) {
 393         return elem.parentNode.removeChild(elem); // Remove a single DOM element
 394       } else if (arguments.length > 1) {
 395         elems = arguments;                        // Multiple parameters
 396       } else if (!tau.isString(elem) && elem.length) {
 397         elems = elem;                             // Single array parameter
 398       }
 399 
 400       // Handles Paramter/Element list removal
 401       removed = [];
 402       if (elems) {
 403         for (i = elems.length - 1; i >= 0; i--) {
 404           e = elems[i];
 405           r = null;
 406           if (e.length) {
 407             r = tau.util.dom.removeElements(e); // Recursively remove a list
 408           } else if (e.parentNode) {
 409             r = e.parentNode.removeChild(e);    // Remove single DOM directly
 410           }
 411 
 412           // Update removed DOM list
 413           if (tau.isArray(r)) {
 414             removed.unshift.apply(removed, r);
 415           } else if (r) {
 416             removed.unshift(r);
 417           }
 418         }
 419       }
 420       return removed;
 421     },

 530     /**
 531      * 편재 페이지를 기반으로 명시된 HTML앨리먼트의 x, y좌표를 반환한다.
 532      * (display:none 이거나 HTML 앨리먼트 트리에 포함되지 않은
 533      * 앨리먼트는 정확한 값을 반환하지 못한다.)
 534      * @param {HTMLElement} elem HTML DOM 앨리먼트
 535      * @return {Array} 명시된 앨리먼트의 XY position [x, y]
 536      */
 537      getXY: function(elem) {
 538          var body = document.getElementById('tau-root'),
 539              dom = parent = elem,
 540              x = y = 0;
 541   
 542          if (!dom || dom === body) {
 543              return [0, 0];
 544          }
 545   
 546          while (parent) {
 547              x += parent.offsetLeft;
 548              y += parent.offsetTop;
 549   
 550              if(parent != dom) {
 551                  // For webkit, we need to add parent's clientLeft/Top as well.
 552                  x += parent.clientLeft || 0;
 553                  y += parent.clientTop || 0;
 554              }
 555   
 556              parent = parent.offsetParent;
 557          }
 558   
 559          // Safari absolute incorrectly account for body offsetTop.
 560          if (tau.rt.isWebkit && tau.util.style(elem, 'position') === 'absolute') {
 561              y -= body.offsetTop;
 562          }
 563   
 564          parent = dom.parentElement;
 565          while (parent && parent != body) {
 566              x -= parent.scrollLeft;
 567              y -= parent.scrollTop;
 568              parent = parent.parentElement;
 569          }
 570          return [x, y];
 571      },

 573      /**
 574       * Element Node(nodeType == 1)를 반환한다. 명시된 노드가
 575       * Element가 아닐경우 parent 중에서 Element Node를 찾아
 576       * 반환한다.
 577       *
 578       * @example
 579       * Node.ELEMENT_NODE == 1
 580       * Node.ATTRIBUTE_NODE == 2
 581       * Node.TEXT_NODE == 3
 582       * Node.CDATA_SECTION_NODE == 4
