  17 /** @lends tau.ui.Controller.prototype */
  18 $class('tau.ui.Controller').extend(tau.rt.EventDelegator).define({
  19   /**
  20    * 생성자, 주어진 Scene객체를 이용하여 Controller객체를 생성한다.
  21    * 
  22    * @class
  23    * 컨트롤러는 {@link tau.ui.Controller.doStart}, 
  24    * {@link tau.ui.Controller.doDestory} 메소드들을 통해
  25    * Scene의 Lifecycle을 관리한다.
  26    * @constructs
  27    * @extends tau.rt.EventDelegator
  28    * @param {Object} [opts] 컨트롤러 생성자 옵션, Optionize를
  29    * 통해 초기 값을 설정할 수 있다.
  30    * @property {Boolean} $isStarted  Controller의 Lifecycle이 시작되었는지
  31    * 상태를 저장한다.
  32    */
  33   Controller: function (opts) {//$ tau.ui.Controller
  34     /** @private Delegators attached to this instance that needs hit testing */
  35     this.$subDelegators = [];
  36     /** @private Controller options */
  37     //this.$opts = {}; // obsolete
  38     //return this;
  39   },
  40 
  41   /**
  42    * Framework 레벨 함수로써 Controller의 Lifecycle을 시작한다.
  43    * Lifecycle이 시작될 때 init메소드를 호출하여 Controller의
  44    * 초기화 작업을 진행 될 수 있도록 한다.
  45    * 이미 Lifecycle이 시작된 Controller는 더이상 initialize작업을
  46    * 수행하지 않는다<p/>
  47    * 
  48    * Controller의 Lifecycle이 시작되었는지 확인을 위해서는
  49    * 현재 인스턴스의 $isStarted property 값을 확인하면
  50    * 된다.<p/>
  51    */
  52   doStart: function () {
  53     if (!this.$isStarted) {
  54       this.init();
  55       this.$isStarted = true;
  56     }
  57   },
  58   
  59   /**
  60    * Framework 레벨 함수로써 Controller의 Lifecycle을 종료하고
  61    * 사용중인 모든 리소스들을 해제한다.
  62    * 이때 내부적으로 destory 메소드가 호출된다.
  63    */
  64   doDestroy: function () {
  65     try {
  66       if (this.$subDelegators) {
  67         for(var i=0, len = this.$subDelegators.length; i < len; i++) {
  68           this.$subDelegators[i].destroy();
  69         }
  70       }
  71       this.destroy();
  72     } finally {
  73       this.unsubscribeEvent();
  74       this.removeBubble();
  75       this.$isStarted = false;
  76       this.$subDelegators = [];
  77       this.$parent = null;
  78     }
  79   },
  80   
  81   /**
  82    * 파라미터로 받은 parent Controller를 현재 Controller의 부모
  83    * Controller로 설정함으로써 Controller의 Hierarchy를 구성한다. </p>
  84    * 주어진 Controller가 부모 Controller로 설정되면 Event Bubbling관계도
  85    * 같이 설정되어 자식 Controller에서 발생된 이벤트는 부모
  86    * Controller에서도 처리가 가능해진다.
  87    * @param {tau.ui.Controller} parent 부모 Controller 객체
  88    * @throws {TypeError} 명시된 파라미터가 tau.ui.Controller의 객체가 아닐 경우
  89    */
  90   setParent: function (parent) {
  91     if (!parent instanceof tau.ui.Controller) {
  92       throw new TypeError(parent + ' is not an instance of tau.ui.Controller: '
  93           + this.currentStack());
  94     }
  95     this.$parent = parent;
  96     if (parent) {
  97       this.setBubble(parent);
  98     }
  99   },
 100 
 101   /**
 102    * 컨트롤러의 부모 컨트롤러를 반환한다.
 103    * @returns {tau.ui.Controller} 부모 컨트롤러 
 104    */
 105   getParent: function () {
 106     return this.$parent;
 107   },
 108 
 109   /**
 110    * 컨트롤러의 제목을 반환한다. 설정되지 않은 경우 클래스명을 사용한다.
 111    * @returns {String} 컨트롤러의 제목, 설정되지 않은 경우 클래스명
 112    */
 113   getTitle: function () {
 114     return this.$title || this.$classname; // default title is its class name
 115   },
 116 
 117   /**
 118    * 컨트롤러의 제목을 설정한다.<p/>
 119    * 주의: {@link tau.ui.ParallelNavigator}를 통해
 120    * Controller를 설정할 경우 title은 반드시 클래스 생성자에서
 121    * 설정되어야 한다. 그렇지 않을 경우 Active 탭이 아닌

 123    * 상태이므로 title이 출력되지 않는 현상이 발생된다.
 124    * @param {String} title 컨트롤러 제목
 125    * @returns {Boolean} 제목이 정상적으로 설정된 경우 true
 126    */
 127   setTitle: function (title) {
 128     if (tau.isString(title)) {
 129       this.$title = title;
 130       return true;
 131     }
 132     return false;
 133   },
 134 
 135   /**
 136    * 컨트롤러에 있는 모든 하위 {@link tau.rt.EventDelegator}을 반환한다. 
 137    * @returns {Array} 하위 EventDelegators를 앨리먼트로 가지는 배열 객체
 138    */
 139   getSubDelegators: function () {
 140     return this.$subDelegators;
 141   },
 142   
 143   /**
 144    * 컨트롤러의 초기화 콜백 함수
 145    * 초기화 수행을 하기 위해서는 오버라이드해야한다.
 146    */
 147   init: function () {
 148   },
 149 
 150   /**
 151    * 컨트롤러의 destroy 콜백 함수. 리소스를 클린업하기 위해 오버라이드해야한다.
 152    */
 153   destroy: function () {
 154   }

 160 /** @lends tau.ui.SceneController.prototype */
 161 $class('tau.ui.SceneController').extend(tau.ui.Controller).define({
 162   /**
 163    * 생성자, 주어진 Scene객체 또는 Scene파일을 이용하여 객체를 생성한다.
 164    * @class
 165    * 사용자화면(scene)의 생성과 사용자로 부터의 반응을
 166    * 처리하기 위한 기능을 정의한다.<p/>
 167    * @constructs
 168    * @extends tau.ui.Controller
 169    * @param {Object} [opts] 컨트롤러 생성자 옵션
 170    */
 171   SceneController: function (opts) {//$ tau.ui.SceneController
 172 
 173     // Prototype(cached) initScene not defined, set to load default named scene 
 174     // file name on next onLoadScene. (Set only on the first contructor call)
 175     // Default scene url: <ctrl path> + <simple ctrl name> + '.scene.js' 
 176     this.$optionize = tau.mixin(this.$optionize, {
 177       handler: {
 178         /** @inner dynamic scene file loading path */
 179         scene: function (prop, scene) {
 180           if (tau.isString(scene) 
 181               && arguments.callee.caller !== this.loadScene) { 
 182             this.clearScene();
 183             if (scene.trim().indexOf('/') == 0) {
 184               this.initScene = tau.getCurrentContext().getRealPath(scene);
 185             } else {
 186               this.initScene = tau.resolveURL(scene);        
 187             }
 188             return true;
 189           } else {
 190             this.setScene(scene);
 191           }
 192         }
 193       }
 194     }, 'remix');
 195   },
 196   
 197   /**
 198    * 이벤트 Subsystem메소드를 Override한 메소드이다.
 199    * SceneController에서 SCENELOADED이벤트가 발생하면 SceneController에 있는 시스템
 200    * 메소드인 doDraw를 호출한다.<p/>
 201    * Scene은 Asynchronous하게 로딩되기 때문에 Scene의 로딩이 완료되기 전에 Draw가 진행
 202    * 되면 화면이 정상적으로 그려지지 않는다. 따라서 <code>tau.rt.Event.SCENELOADED</code>
 203    * 이벤트를 받은 다음 이후 과정인 Draw를 수행해야 한다.
 204    * @param {tau.rt.Event} e Mobello 이벤트 객체
 205    */
 206   propagateEvent: function (e) {
 207     switch (e.getName()) {
 208       case tau.rt.Event.SCENELOADED: // async call
 209         this.doDraw();
 210         break;
 211     }
 212     tau.ui.SceneController.$super.propagateEvent.apply(this, arguments);
 213   },
 214   
 215   /**
 216    * Controller Lifecycle 메소드를 Override한다. initialize 작업을 완료한 후에
 217    * Scene이 로딩되어 있지 않다면 Scene 로딩 작업을 수행한다. 이때 수행되는 Scene 로딩
 218    * 작업은 Asynchronous하게 진행된다.
 219    * @see tau.ui.Controller.doStart
 220    */
 221   doStart: function () {
 222     tau.ui.SceneController.$super.doStart.apply(this, arguments);
 223     this.loadScene(); // load scene asynchronously if so, it returns no delay
 224     var comps = this.getScene().getComponents();
 225     if (comps && comps.length > 0) { // user loaded scene programmatically
 226       this.sceneLoaded();
 227       this.doDraw();
 228     }
 229   },
 230   
 231   /**
 232    * SceneController가 소유하고 있는 Scene의 Drawing작업을 수행한다. 먼저 Drawing
 233    * 작업을 수행하기 전에 Scene에 정의된 컴포넌트들이 모듀 Loading되어 있어야 하며 이 작업은
 234    * Asynchronous하게 진행된다.<p/>
 235    * Drawing작업이 완려되면 <code>tau.rt.Event.SCENEDRAWN</code> 이벤트가 발생된다.
 236    */
 237   doDraw: function () {
 238     var parent = this.getParent();
 239     if (this.getScene().draw(parent.getDOM(tau.ui.CONTENT_KEY))) {
 240       this.sceneDrawn();
 241       var e = new tau.rt.Event(tau.rt.Event.SCENEDRAWN, this);
 242       e.alwaysBubble();
 243       this.fireEvent(e);
 244     }
 245   },
 246 
 247   /**
 248    * 현재 컨트롤러에 설정된 Navigation Bar를 반환한다.

 305   /**
 306    * 컨트롤러가 관리하는 scene을 반환한다.
 307    * @returns {tau.ui.Scene} 컨트롤러가 관리하는 scene 객체

 309   getScene: function () {
 310     if (!(this.$scene instanceof tau.ui.Scene)) {

 316   /**
 317    * 컨트롤러에{@link tau.ui.Scene}을 설정한다.
 318    * @param {Object} scene 설정할 scene 객체
 319    * @returns {Boolean} scene이 설정되었으면 true
 320    */
 321   setScene: function (scene) {
 322     if (scene instanceof tau.ui.Scene && this.$scene !== scene) {
 323       this.clearScene();
 324       tau.arr(this.$subDelegators).pushUnique(scene);
 325       this.$scene = scene;
 326       return true;
 327     }
 328     return false;
 329   },

 338       this.$scene = null;
 339     }

 360   /**
 361    * Controller가 관리하는 Scene을 로드한다
 362    * 개발자가 직접 프로그래밍을 통해 Scene을 생성하기 위해서는
 363    * 이 메소드를 Override해서 컴포넌트를 통해 직접 화면을
 364    * 구성해야 한다.
 365    * <p/>
 366    * 먄약 NamingConvention에 따라 SceneController가 있는 
 367    * 폴더 내에 Scene파일과 CS(Compiled Scene)파일이
 368    * 생성되었다면 SceneController의 loadScene() 메소드를
 369    * 구현하지 않아도 자동으로 해당 CS파일을 로딩하여 Scene을 구성한다.
 370    * <p/>
 371    * NamingConvention에 의해 자동으로 Scene을 로딩하지 못하는
 372    * 경우 즉, 해당 CS파일이 다른 폴더에 있거나 아니면
 373    * NamingConvention이 다를 경우 SceneController의 인스턴스를
 374    * 생성할 때 로딩하고자 하는 CS파일을 직접 지정함으로써 편리하게
 375    * 해당 Scene을 로딩할 수도 있다.<p/>
 376    *
 377    * @example
 378    * // Programming으로 직접 Scene 생성
 379    * $class('sample.FooController').extend(tau.ui.SceneController).define({
 380    *   ...
 381    *
 382    *   loadScene: function () { // creates scene manually
 383    *     var label, button,
 384    *         scene = this.getScene();
 385    *     label = new tau.ui.Label({text: 'Name: '});
 386    *     scene.add(label);
 387    *     button = new tau.ui.Button({label: 'Mobello'});
 388    *     scene.add(button);
 389    *   }
 390    *
 391    *   ...
 392    * });
 393    *
 394    * @example
 395    * // Naming Convention에 의한 Scene Loading
 396    * // SceneController 클래스 명
 397    * $class('sample.FooController').extend(tau.ui.SceneController).define({
 398    *   ...
 399    * });
 400    *
 401    * // Scene파일 명
 402    * foo.scene
 403    *
 404    * // CS(Compiled Scene)파일 명
 405    * foo.scene.js
 406    *
 407    * @example
 408    * // 직접 CS파일을 지정하여 Scene을 Loading
 409    * // FooController에서 'dir'폴더에 있는 'bar.scene.js'
 410    * // CS파일을 로딩하여 Scene을 구성
 411    *
 412    * var ctrl = new sample.FooController({scene: '/dir/bar.scene.js'});
 413    */
 414   loadScene: function () {
 415     var classname, clazz, filepath,
 416         that = this; 
 417     
 418     function _loadScene() { //fallback method
 419       if  (tau.isFunction(that.initScene)) {
 420         that.initScene();
 421       }
 422       that.sceneLoaded();
 423       that.doDraw();
 424     }
 425     if (tau.isFunction(that.initScene)) {
 426       that.initScene(); // synchronous
 427     } else {
 428       classname = that.$classname;
 429       clazz = $class.forName(classname);
 430       filepath = clazz.$filepath;
 431       if (!tau.isFunction(that.initScene) 
 432           && !this.hasOwnProperty('initScene')) {
 433         clazz.prototype.initScene = 
 434           filepath.substring(0, filepath.lastIndexOf('/') + 1)
 435             + classname.substring(classname.lastIndexOf('.') + 1)
 436               .replace(/[Cc]ontroller$/, '').toLowerCase() + '.scene.js';
 437       }
 438       new tau.ScriptHelper({
 439         callbackFn: function () {
 440           tau.log.debug('Scene file loaded: ' + that.initScene, that);
 441           if (tau.isString(clazz.prototype.initScene)) {
 442             clazz.prototype.initScene = window.initScene;
 443           } else {
 444             that.initScene = window.initScene;
 445           }
 446           window.initScene = undefined; // delete imported initScene function
 447           _loadScene();
 448         },
 449         timeoutFn: function () {
 450           tau.log.debug('Scene file is unavailable: ' + that.initScene, that);
 451           _loadScene();
 452         }
 453       }).load(that.initScene);
 454     }
 455   },

 477   /**
 478    * 명시된 콘트롤러(ctrl)를 이용하여 Modal화면을 생성한다.<p/>
 479    * Modal화면을 생성할때 다양한 옵션을 통해 Modal화면의
 480    * 특성을 조정할 수 있다.
 481    * @example
 482    * $class('sample.FooController').extend(tau.ui.SceneController).define({
 483    *   loadScene: function () {
 484    *     var btn = new tau.ui.Button({label: 'Show modal!'});
 485    *     btn.onEvent(tau.rt.Event.TAP, this.handleClick, this);
 486    *     this.getScene().add(btn);
 487    *   },
 488    *
 489    *   handleClick: function(e, payload) {
 490    *     var modalCtrl = new sample.ModalController();
 491    *     // presents modal scene
 492    *     this.presentModal(modalCtrl, {layout: 'FORM', 'animate': 'vertical'});
 493    *   }
 494    * });
 495    * @param {tau.ui.Controller} ctrl tau.ui.Controller를
 496    * 상속받은 클래스
 497    * @param {Object} [opts] Modal화면의 생성 옵션
 498    * @param {String} [opts.layout = 'FULL'] Modal화면의 형태를 나타내며
 499    * 'FULL', 'PAGE', 'FORM' 중의 하나를 사용할 수 있다. 
 500    * @param {String} [opts.animate = 'vertical'] Animation을

 792   /**

 795    * @param {Object} modal Modal 객체

 797   adjustDimension: function (modal) {
 798     var top = left = '0';
 799     if (!modal) {
 800       modal = this.peekModal();
 801     }
 802     if (!modal) return;
 803 
 804     var dimension, width, height,
 805         dom = modal.ctrl.getDOM(),
 806         layout = 'FULL';
 807

 810     }
 811     dimension = tau.ui.ModalSceneController['LAYOUT_' + layout];
 812     width = dimension.width;
 813     height = dimension.height;
 814     if (width !== '100') {
 815       left = (tau.getWidth() - width) / 2;
 816     }
 817     if (height !== '100') {
 818       top = (tau.getHeight() - height) / 2;
 819     }
 820     dom.style.removeProperty('-webkit-transform');
 821     dom.style.top = top + 'px';
 822     dom.style.left = left + 'px';
 823     dom.style.width = width + (tau.isString(width) ? '%' : 'px');
 824     dom.style.height = height + (tau.isString(height) ? '%' : 'px');
 825   },

 840    * layout과 animation타입을 설정할 수 있다. 자세한 내용은
 841    * {@link tau.ui.SceneController.presentModal}을 참조하면
 842    * 된다.
 843    * @param {tau.ui.Controller} ctrl {@link tau.ui.Controller} 객체
 844    * @param {Object} [opts] JSON객체로 표현하며 layout과 animation타입을 설정한다.
 845    * @param {String} [opts.layout = 'FULL'] Modal화면의 형태를 나타내며
 846    * 'FULL', 'PAGE', 'FORM' 중의 하나를 사용할 수 있다.
 847    * @param {String} [opts.animate = 'vertical'] Animation을 통해 Modal화면이
 848    * 출력되는 방향을 나타낸다. 'vertical' 또는 'horizontal'의 값을 사용할 수 있다.
 849    * @see tau.ui.SceneController#presentModal
 850    */
 851   present: function (ctrl, opts) {
 852     if (!this.hasModals()) {
 853       this.dim();
 854     }
 855     var dom,
 856         old = this.peekModal(),
 857         zIndex = old ? old.ctrl.getDOM().style.zIndex : this._zIndex;
 858     zIndex = window.parseInt(zIndex) + 1;
 859     this.pushModal({'ctrl': ctrl, 'opts': opts});
 860     ctrl.setParent(this);
 861     // animation will be processed in handleSceneDrawn method
 862     dom = ctrl.getDOM();
 863     dom.style.zIndex = zIndex;
 864     dom.style.backgroundColor = '#ffffff';
 865     this.adjustDimension();
 866     if (opts && opts.animate) {
 867       dom.style.visibility = 'hidden';
 868     }
 869     ctrl.doStart();
 870   },

 872   /**
 873    * 현재 출력되어 있는 Modal Scene을 제거한 다음 Modal Scene을 시작한
 874    * Scene으로 돌아가거나 이전 ModalScene상태로
 875    * 되돌아간다.<p/>
 876    * 만약 여러개의 SceneController를 Modal로 띄워졌을 경우
 877    * 최상위에 있는 SceneController를 사라지게 한다.
 878    * @param {tau.ui.Controller} ctrl Modal Scene을 생성하기 위한 Controller
 879    * @param {Boolean} animated true이면 Animation이 적용된다.
 880    */
 881   dismiss: function (ctrl, animate) {
 882     var tobe = this.peekModal(1);
 883     if (tobe) { // reveals to be modal scene!
 884       this.adjustDimension(tobe);
 885       tobe.ctrl.getDOM().style.display = '';
 886     }
 887     var current = this.peekModal();
 888     // TODO dismiss if current is equal to ctrl(parameter)
 889     if (animate && current.opts && current.opts.animate) {
 890       this._shieldSystemEvents(true);
 891       if (current.opts.animate) {
 892         try {
 893           this.animate(current.opts.animate, false, tau.ctxAware(cleanup, this));
 894         } catch (e) {
 895           this._shieldSystemEvents(false);
 896           tau.log.error('Can not animate: ' + e);
 897         }
 898       }
 899     } else {
 900       cleanup.call(this);
 901     }
 902     /** @inner */
 903     function cleanup() {
 904       var popped = this.popModal();
 905       popped.ctrl.setParent(null);
 906       popped.ctrl.removeBubble(this);
 907       try {
 908         popped.ctrl.doDestroy();
 909       } finally {
 910         this._shieldSystemEvents(false);
 911       }
 912       if (!this.hasModals()) {
 913         this.undim();
 914       }
 915     }
 916   },

 918   /**

 928   /**
 929    * Animation효과를 처리한다. Modal Scene이 밑에서 부터 위로
 930    * 출력되고 반대방향으로 사라지도록 한다.
 931    * @param {Boolean} visible true이면 ModalScene이 출력되며 false일 경우
 932    * 사라지도록 처리한다.
 933    * @param {Function} [callbackFn] Animation이 종료되면 호출되는 callback함수
 934    * @see tau.ui.ModalSceneController#animate
 935    */
 936   transitVertical: function (visible, callbackFn) {
 937     var anim,
 938         dom = this.peekModal().ctrl.getDOM(),
 939         px = dom.style.top.indexOf('px'),
 940         height = offset = tau.getHeight();
 941     if (px != -1) {
 942       offset = height - window.parseInt(dom.style.top.substring(0, px));
 943       if (visible) {
 944         offset = '-'.concat(offset);
 945       }
 946     }
 947     if (visible) {
 948       dom.style.top = height + 'px';
 949       dom.style.visibility = 'visible';
 950     }
 951     anim = new tau.fx.Transition({
 952       timingFunction: 'ease-in-out',
 953       duration: tau.rt.hasTouch ? '0.25s' : '0.5s'
 954     });
 955     
 956     anim.setStyle('-webkit-transform', 'translateY(' + offset + 'px)', {
 957       onEnd: callbackFn
 958     });
 959     anim.animate(dom);
 960   },

1408

1410           var content = that.getDOM(tau.ui.CONTENT_KEY);
1411           content.removeChild(popped.ctrl.getDOM());
1412           if (popped.nav) popped.nav.destroy();
1413           if (popped.tool) popped.tool.destroy();
1414           /** @inner Delayed cleanup for slow controllers (e.g. big Table) */
1415           window.setTimeout(function () {
1416             popped.ctrl.doDestroy();
1417           }, 500);
1418           popped.opts.poppedFn();
1419

1421           that.fireEvent(e, {'fg': active, 'bg': popped});
