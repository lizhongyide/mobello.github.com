var scene = this.getScene();
var panel = new tau.ui.Panel({
  width: '100px',
  height: '100px'
});
var innerPanel1 = new tau.ui.Panel({
  width: '100px',
  height: '50px',
  backgroundColor: 'blue'
});
var innerPanel2 = new tau.ui.Panel({
  width: '100px',
  height: '100px',
  backgroundColor: 'red'
});
panel.setComponents([innerPanel1, innerPanel2]);
scene.add(panel);
