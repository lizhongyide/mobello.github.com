Class tau.ui.ScrollPanel

  • HIERARCHY
    tau.TObject
   |

         +--tau.ui.ScrollPanel
  • All Mixin Classes:
    tau.ui.Container
  • Direct Known Subclasses:
    tau.ui.Carousel, tau.ui.Dialog, tau.ui.Table, tau.ui.TextArea, tau.ui.TextView, tau.ui.ToolBar
  • Defined in:
    tau.ui.comp.js

ScrollPanel은 인스턴스가 포함되어 있는 tau.ui.Scene의 크기보다 큰 컨텐츠를 화면에 보여지도록 스크롤을 제공하는 클래스이다.

기본적으로 자동으로 컨텐트가 사이즈보다 크게 되면 수직 스크롤 혹은 수평 스크롤을 지원하는 컨테이너 클래스이다.

대표적인 기능 :

  • tau.ui.ScrollPanel.vScroll: 수직 스크롤을 기본으로 지원한다.
  • tau.ui.ScrollPanel.hScroll: 수평 스크롤을 기본으로 지원한다.
  • tau.ui.ScrollPanel.vScrollbar: 수직 스크롤바 표시 여부 지원
  • tau.ui.ScrollPanel.hScrollbar: 수평 스크롤바 표시 여부 지원
  • tau.ui.ScrollPanel.momentum: 가속도 지원
  • tau.ui.ScrollPanel.snap: 페이지 단위 스크롤 지원
  • tau.ui.ScrollPanel.pullToRefresh: pull to refresh(양맨끝, 양쪽끝을 밀었다 당기는 기능)

다음 사항에 대해서는 주의해야한다. 하위에 동일한 방향의 스크롤기능이 있는 ScrollPanel, tau.ui.Slider, tau.ui.Switch이 있는 경우 하위에 있는 컴포넌트에서 이벤트를 bubble up되지 않도록 처리하는 것이 좋다. 

 
loadScene: function () {
   var scrollPanel = new tau.ui.ScrollPanel({
     styles: {height: '300px', width: '100px'}
   });
   var innerPanel = tau.ui.ScrollPanel({
     styles: {height: '200px', width: '100px'},
     components: [
       new tau.ui.Button({styles: {height: '100px', width: '100px'}}),
       new tau.ui.Button({styles: {height: '100px', width: '100px'}}),
       new tau.ui.Button({styles: {height: '100px', width: '100px'}}),
       new tau.ui.Button({styles: {height: '100px', width: '100px'}})
     ]
   });
   var slider = new tau.ui.Slider({
     vertical: true,
     styles: {height: '200px'}
   });
 
   innerPanel.onEvent(tau.rt.Event.TOUCHSTART, this.handleTouchStart, this);
   slider.onEvent(tau.rt.Event.TOUCHSTART, this.handleTouchStart, this);
 
   scrollPanel.setComponents([innerPanel, slider]);
 },
 
 handleTouchStart: function (e, payload) {
   e.stopPropagation(); // Bubble up단계에 등록된 이벤트 리스너는 동작하지 않는다.
 }

// pull to refresh
loadScene: function () {
  var scene = this.getScene();
  var scrollPanel = new tau.ui.ScrollPanel({
    id: 'scrollPanel',
    pullToRefresh: 'down',
    pullDownLabel: ['업데이트하시려면 아래로 당기세요.', '업데이트하시려면 당겼다 놓으세요.', '업데이트중...'],
    components: this.getComponents()
  });
  
  scrollPanel.onEvent('pullToRefreshUpdate', this.handleLoad, this);
},

getComponents: function () {
  var comps = [];
  for(var i = 0; i < 5; i++ ) {
    comp[i] = new tau.ui.Button({styles: {width: '100px', height: '100px'}});
  }
  return comps;
},

handleLoad: function (e, payload) {
  var scene = this.getScene();
  var scrollPanel = scene.getComponent('scrollPanel');
  var comps = this.getComponents(), len = comps.len;
  
  for(var i=0; i < len; i++) {
    scrollPanel.add(comps[i], 0); // 맨 위에 추가되도록 처리한다.
  }
}
